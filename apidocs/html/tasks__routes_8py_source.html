156 @router.get("/activity/", response_model=List[tasks_schemas.TaskHistoryCount])
157 async def task_activity(
158  project_id: int, days: int = 10, db: Session = Depends(database.get_db)
159 ):
160  """Retrieves the validate and mapped task count for a specific project.
161 
162  Args:
163  project_id (int): The ID of the project.
164  days (int): The number of days to consider for the
165  task activity (default: 10).
166  db (Session): The database session.
167 
168  Returns:
169  list[TaskHistoryCount]: A list of task history counts.
170 
171  """
172  end_date = datetime.now() - timedelta(days=days)
173  task_list = await tasks_crud.get_task_id_list(db, project_id)
174  tasks = []
175  for task_id in task_list:
176  tasks.extend(
177  [tasks_crud.get_project_task_history(task_id, False, end_date, db)]
178  )
179  task_history = await asyncio.gather(*tasks)
180  return await tasks_crud.count_validated_and_mapped_tasks(
181  task_history,
182  end_date,
183  )
