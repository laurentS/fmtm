130 async def geojson_to_flatgeobuf(
131  db: Session, geojson: geojson.FeatureCollection
132 ) -> Optional[bytes]:
133  """From a given FeatureCollection, return a memory flatgeobuf obj.
134 
135  NOTE this generate an fgb with string timestamps, not datetime.
136  NOTE ogr2ogr would generate datetime, but parsing does not seem to work.
137 
138  Args:
139  db (Session): SQLAlchemy db session.
140  geojson (geojson.FeatureCollection): a FeatureCollection object.
141 
142  Returns:
143  flatgeobuf (bytes): a Python bytes representation of a flatgeobuf file.
144  """
145  geojson_with_props = add_required_geojson_properties(geojson)
146 
147  sql = text(
148  """
149  DROP TABLE IF EXISTS temp_features CASCADE;
150 
151  -- Wrap geometries in GeometryCollection
152  CREATE TEMP TABLE IF NOT EXISTS temp_features(
153  geom geometry(GeometryCollection, 4326),
154  osm_id integer,
155  tags text,
156  version integer,
157  changeset integer,
158  timestamp text
159  );
160 
161  WITH data AS (SELECT CAST(:geojson AS json) AS fc)
162  INSERT INTO temp_features
163  (geom, osm_id, tags, version, changeset, timestamp)
164  SELECT
165  ST_ForceCollection(ST_GeomFromGeoJSON(feat->>'geometry')) AS geom,
166  regexp_replace(
167  (feat->'properties'->>'osm_id')::text, '[^0-9]', '', 'g'
168  )::integer as osm_id,
169  (feat->'properties'->>'tags')::text as tags,
170  (feat->'properties'->>'version')::integer as version,
171  (feat->'properties'->>'changeset')::integer as changeset,
172  (feat->'properties'->>'timestamp')::text as timestamp
173  FROM json_array_elements((SELECT fc->'features' FROM data)) AS f(feat);
174 
175  -- Second param = generate with spatial index
176  SELECT ST_AsFlatGeobuf(geoms, true)
177  FROM (SELECT * FROM temp_features) AS geoms;
178  """
179  )
180 
181  # Run the SQL
182  result = db.execute(sql, {"geojson": json.dumps(geojson_with_props)})
183  # Get a memoryview object, then extract to Bytes
184  flatgeobuf = result.first()
185 
186  if flatgeobuf:
187  return flatgeobuf[0].tobytes()
188 
189  # Nothing returned (either no features passed, or failed)
190  return None

261 async def split_geojson_by_task_areas(
262  db: Session,
263  featcol: geojson.FeatureCollection,
264  project_id: int,
265 ) -> Optional[dict[int, geojson.FeatureCollection]]:
266  """Split GeoJSON into tagged task area GeoJSONs.
267 
268  NOTE inserts feature.properties.osm_id as feature.id for each feature.
269  NOTE ST_Within used on polygon centroids to correctly capture the geoms per task.
270 
271  Args:
272  db (Session): SQLAlchemy db session.
273  featcol (bytes): Data extract feature collection.
274  project_id (int): The project ID for associated tasks.
275 
276  Returns:
277  dict[int, geojson.FeatureCollection]: {task_id: FeatureCollection} mapping.
278  """
279  sql = text(
280  """
281  -- Drop table if already exists
282  DROP TABLE IF EXISTS temp_features CASCADE;
283 
284  -- Create a temporary table to store the parsed GeoJSON features
285  CREATE TEMP TABLE temp_features (
286  id VARCHAR,
287  geometry GEOMETRY,
288  properties JSONB
289  );
290 
291  -- Insert parsed geometries and properties into the temporary table
292  INSERT INTO temp_features (id, geometry, properties)
293  SELECT
294  (feature->'properties'->>'osm_id')::VARCHAR AS id,
295  ST_SetSRID(ST_GeomFromGeoJSON(feature->>'geometry'), 4326) AS geometry,
296  jsonb_set(
297  jsonb_set(
298  feature->'properties',
299  '{task_id}', to_jsonb(tasks.project_task_index), true
300  ),
301  '{project_id}', to_jsonb(tasks.project_id), true
302  ) AS properties
303  FROM (
304  SELECT jsonb_array_elements(CAST(:geojson_featcol AS jsonb)->'features')
305  AS feature
306  ) AS features
307  JOIN tasks ON tasks.project_id = :project_id
308  WHERE
309  ST_Within(
310  ST_Centroid(ST_SetSRID(ST_GeomFromGeoJSON(feature->>'geometry'), 4326)
311  ), tasks.outline);
312 
313  -- Retrieve task outlines based on the provided project_id
314  SELECT
315  tasks.project_task_index AS task_id,
316  jsonb_build_object(
317  'type', 'FeatureCollection',
318  'features', jsonb_agg(feature)
319  ) AS task_features
320  FROM
321  tasks
322  LEFT JOIN LATERAL (
323  SELECT
324  jsonb_build_object(
325  'type', 'Feature',
326  'geometry', ST_AsGeoJSON(temp_features.geometry)::jsonb,
327  'id', temp_features.id,
328  'properties', temp_features.properties
329  ) AS feature
330  FROM (
331  SELECT DISTINCT ON (geometry)
332  id,
333  geometry,
334  properties
335  FROM temp_features
336  ) AS temp_features
337  WHERE
338  ST_Within(ST_Centroid(temp_features.geometry), tasks.outline)
339  ) AS feature ON true
340  WHERE
341  tasks.project_id = :project_id
342  GROUP BY
343  tasks.project_task_index;
344  """
345  )
346 
347  try:
348  result = db.execute(
349  sql,
350  {
351  "geojson_featcol": json.dumps(featcol),
352  "project_id": project_id,
353  },
354  )
355  feature_collections = result.all()
356 
357  except ProgrammingError as e:
358  log.error(e)
359  log.error("Attempted geojson task splitting failed")
360  return None
361 
362  if feature_collections:
363  # NOTE the feature collections are nested in a tuple, first remove
364  task_geojson_dict = {
365  record[0]: geojson.loads(json.dumps(record[1]))
366  for record in feature_collections
367  }
368  return task_geojson_dict
369 
370  return None

373 def add_required_geojson_properties(
374  geojson: geojson.FeatureCollection,
375 ) -> geojson.FeatureCollection:
376  """Add required geojson properties if not present.
377 
378  This step is required prior to flatgeobuf generation,
379  else the workflows of conversion between the formats will fail.
380  """
381  for feature in geojson.get("features", []):
382  properties = feature.get("properties", {})
383 
384  # The top level id is defined, set to osm_id
385  if feature_id := feature.get("id"):
386  properties["osm_id"] = feature_id
387 
388  # Check for id type embedded in properties
389  if osm_id := properties.get("osm_id"):
390  # osm_id property exists, set top level id
391  feature["id"] = osm_id
392  else:
393  if prop_id := properties.get("id"):
394  # id is nested in properties, use that
395  feature["id"] = prop_id
396  properties["osm_id"] = prop_id
397  elif fid := properties.get("fid"):
398  # The default from QGIS
399  feature["id"] = fid
400  properties["osm_id"] = fid
401  else:
402  # Random id
403  # NOTE 32-bit int is max supported by standard postgres Integer
404  random_id = getrandbits(30)
405  feature["id"] = random_id
406  properties["osm_id"] = random_id
407 
408  # Other required fields
409  if not properties.get("tags"):
410  properties["tags"] = ""
411  if not properties.get("version"):
412  properties["version"] = 1
413  if not properties.get("changeset"):
414  properties["changeset"] = 1
415  if not properties.get("timestamp"):
416  properties["timestamp"] = timestamp().strftime("%Y-%m-%dT%H:%M:%S")
417 
418  return geojson

421 def parse_and_filter_geojson(
422  geojson_raw: Union[str, bytes], filter: bool = True
423 ) -> Optional[geojson.FeatureCollection]:
424  """Parse geojson string and filter out incompatible geometries."""
425  geojson_parsed = geojson.loads(geojson_raw)
426 
427  if isinstance(geojson_parsed, geojson.FeatureCollection):
428  log.debug("Already in FeatureCollection format, skipping reparse")
429  featcol = geojson_parsed
430  elif isinstance(geojson_parsed, geojson.Feature):
431  log.debug("Converting Feature to FeatureCollection")
432  featcol = geojson.FeatureCollection(features=[geojson_parsed])
433  else:
434  log.debug("Converting Geometry to FeatureCollection")
435  featcol = geojson.FeatureCollection(
436  features=[geojson.Feature(geometry=geojson_parsed)]
437  )
438 
439  # Exit early if no geoms
440  if not (features := featcol.get("features", [])):
441  return None
442 
443  # Strip out GeometryCollection wrappers
444  for feat in features:
445  geom = feat.get("geometry")
446  if (
447  geom.get("type") == "GeometryCollection"
448  and len(geom.get("geometries")) == 1
449  ):
450  feat["geometry"] = geom.get("geometries")[0]
451 
452  # Return unfiltered featcol
453  if not filter:
454  return featcol
455 
456  # Filter out geoms not matching main type
457  geom_type = get_featcol_main_geom_type(featcol)
458  features_filtered = [
459  feature
460  for feature in features
461  if feature.get("geometry", {}).get("type", "") == geom_type
462  ]
463 
464  return geojson.FeatureCollection(features_filtered)

467 def get_featcol_main_geom_type(featcol: geojson.FeatureCollection) -> str:
468  """Get the predominant geometry type in a FeatureCollection."""
469  geometry_counts = {"Polygon": 0, "Point": 0, "Polyline": 0}
470 
471  for feature in featcol.get("features", []):
472  geometry_type = feature.get("geometry", {}).get("type", "")
473  if geometry_type in geometry_counts:
474  geometry_counts[geometry_type] += 1
475 
476  return max(geometry_counts, key=geometry_counts.get)

479 async def check_crs(input_geojson: Union[dict, geojson.FeatureCollection]):
480  """Validate CRS is valid for a geojson."""
481  log.debug("validating coordinate reference system")
482 
483  def is_valid_crs(crs_name):
484  valid_crs_list = [
485  "urn:ogc:def:crs:OGC:1.3:CRS84",
486  "urn:ogc:def:crs:EPSG::4326",
487  "WGS 84",
488  ]
489  return crs_name in valid_crs_list
490 
491  def is_valid_coordinate(coord):
492  if coord is None:
493  return False
494  return -180 <= coord[0] <= 180 and -90 <= coord[1] <= 90
495 
496  error_message = (
497  "ERROR: Unsupported coordinate system, it is recommended to use a "
498  "GeoJSON file in WGS84(EPSG 4326) standard."
499  )
500  if "crs" in input_geojson:
501  crs = input_geojson.get("crs", {}).get("properties", {}).get("name")
502  if not is_valid_crs(crs):
503  log.error(error_message)
504  raise HTTPException(status_code=400, detail=error_message)
505  return
506 
507  if (input_geojson_type := input_geojson.get("type")) == "FeatureCollection":
508  features = input_geojson.get("features", [])
509  coordinates = (
510  features[-1].get("geometry", {}).get("coordinates", []) if features else []
511  )
512  elif input_geojson_type == "Feature":
513  coordinates = input_geojson.get("geometry", {}).get("coordinates", [])
514  else:
515  coordinates = input_geojson.get("coordinates", {})
516 
517  first_coordinate = None
518  if coordinates:
519  while isinstance(coordinates, list):
520  first_coordinate = coordinates
521  coordinates = coordinates[0]
522 
523  if not is_valid_coordinate(first_coordinate):
524  log.error(error_message)
525  raise HTTPException(status_code=400, detail=error_message)

574 async def geojson_to_javarosa_geom(geojson_geometry: dict) -> str:
575  """Convert a GeoJSON geometry to JavaRosa format string.
576 
577  This format is unique to ODK and the JavaRosa XForm processing library.
578  Example JavaRosa polygon (semicolon separated):
579  -8.38071535576881 115.640801902838 0.0 0.0;
580  -8.38074220774489 115.640848633963 0.0 0.0;
581  -8.38080128208577 115.640815355738 0.0 0.0;
582  -8.38077407987063 115.640767444534 0.0 0.0;
583  -8.38071535576881 115.640801902838 0.0 0.0
584 
585  Args:
586  geojson_geometry (dict): The GeoJSON geometry.
587 
588  Returns:
589  str: A string representing the geometry in JavaRosa format.
590  """
591  if geojson_geometry is None:
592  return ""
593 
594  coordinates = []
595  if geojson_geometry["type"] in ["Point", "LineString", "MultiPoint"]:
596  coordinates = [[geojson_geometry.get("coordinates", [])]]
597  elif geojson_geometry["type"] in ["Polygon", "MultiLineString"]:
598  coordinates = geojson_geometry.get("coordinates", [])
599  elif geojson_geometry["type"] == "MultiPolygon":
600  # Flatten the list structure to get coordinates of all polygons
601  coordinates = sum(geojson_geometry.get("coordinates", []), [])
602  else:
603  raise ValueError("Unsupported GeoJSON geometry type")
604 
605  javarosa_geometry = []
606  for polygon in coordinates:
607  for lon, lat in polygon:
608  javarosa_geometry.append(f"{lat} {lon} 0.0 0.0")
609 
610  return ";".join(javarosa_geometry)

613 async def javarosa_to_geojson_geom(javarosa_geom_string: str, geom_type: str) -> dict:
614  """Convert a JavaRosa format string to GeoJSON geometry.
615 
616  Args:
617  javarosa_geom_string (str): The JavaRosa geometry.
618  geom_type (str): The geometry type.
619 
620  Returns:
621  dict: A geojson geometry.
622  """
623  if javarosa_geom_string is None:
624  return {}
625 
626  if geom_type == "Point":
627  lat, lon, _, _ = map(float, javarosa_geom_string.split())
628  geojson_geometry = {"type": "Point", "coordinates": [lon, lat]}
629  elif geom_type == "Polyline":
630  coordinates = [
631  [float(coord) for coord in reversed(point.split()[:2])]
632  for point in javarosa_geom_string.split(";")
633  ]
634  geojson_geometry = {"type": "LineString", "coordinates": coordinates}
635  elif geom_type == "Polygon":
636  coordinates = [
637  [
638  [float(coord) for coord in reversed(point.split()[:2])]
639  for point in coordinate.split(";")
640  ]
641  for coordinate in javarosa_geom_string.split(",")
642  ]
643  geojson_geometry = {"type": "Polygon", "coordinates": coordinates}
644  else:
645  raise ValueError("Unsupported GeoJSON geometry type")
646 
647  return geojson_geometry

686 def multipolygon_to_polygon(features: Union[Feature, FeatCol, MultiPolygon, Polygon]):
687  """Converts a GeoJSON FeatureCollection of MultiPolygons to Polygons.
688 
689  Args:
690  features : A GeoJSON FeatureCollection containing MultiPolygons/Polygons.
691 
692  Returns:
693  geojson.FeatureCollection: A GeoJSON FeatureCollection containing Polygons.
694  """
695  geojson_feature = []
696  features = parse_featcol(features)
697 
698  # handles both collection or single feature
699  features = features.get("features", [features])
700 
701  for feature in features:
702  properties = feature["properties"]
703  geom = shape(feature["geometry"])
704  if geom.geom_type == "Polygon":
705  geojson_feature.append(
706  geojson.Feature(geometry=geom, properties=properties)
