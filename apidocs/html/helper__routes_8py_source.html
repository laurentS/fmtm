154 @router.post("/create-entities-from-csv")
155 async def create_entities_from_csv(
156  csv_file: UploadFile,
157  odk_project_id: int,
158  entity_name: str,
159  odk_creds: ODKCentral = Depends(),
160  current_user: AuthUser = Depends(login_required),
161 ):
162  """Upload a CSV file to create new ODK Entities in a project.
163 
164  The Entity must already be defined on the server.
165  The CSV fields must match the Entity fields.
166  """
167  filename = Path(csv_file.filename)
168  file_ext = filename.suffix.lower()
169 
170  if file_ext != ".csv":
171  raise HTTPException(
172  status_code=HTTPStatus.BAD_REQUEST, detail="Provide a valid .csv"
173  )
174 
175  def parse_csv(csv_bytes):
176  parsed_data = []
177  csv_str = csv_bytes.decode("utf-8")
178  csv_reader = csv.DictReader(StringIO(csv_str))
179  for row in csv_reader:
180  parsed_data.append(dict(row))
181  return parsed_data
182 
183  parsed_data = parse_csv(await csv_file.read())
184  entities_data_dict = {str(uuid4()): data for data in parsed_data}
185 
186  async with central_deps.get_odk_entity(odk_creds) as odk_central:
187  entities = await odk_central.createEntities(
188  odk_project_id,
189  entity_name,
190  entities_data_dict,
191  )
192 
193  return entities
