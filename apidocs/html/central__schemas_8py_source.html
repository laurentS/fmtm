128  @field_validator("label", mode="before")
129  @classmethod
130  def append_status_emoji(cls, value: str, info: ValidationInfo) -> str:
131  """Add 🔒 (locked), ✅ (complete) or ❌ (invalid) emojis."""
132  status = info.data.get("status", TaskStatus.READY.value)
133  emojis = {
134  str(TaskStatus.LOCKED_FOR_MAPPING.value): "🔒",
135  str(TaskStatus.MAPPED.value): "✅",
136  str(TaskStatus.INVALIDATED.value): "❌",
137  str(TaskStatus.BAD.value): "❌",
138  }
139 
140  # Remove any existing emoji at the start of the label
141  for emoji in emojis.values():
142  if value.startswith(emoji):
143  value = value[len(emoji) :].lstrip()
144  break
145 
146  if status in emojis.keys():
147  value = f"{emojis[status]} {value}"
148 
149  return value
