153  """Update or create new submission JSON in S3 for a project."""
154  try:
155  # Get Project

163 
164  if not odk_forms:
165  msg = f"No odk forms returned for project ({project_id})"
166  log.warning(msg)
167  return HTTPException(
168  status_code=HTTPStatus.UNPROCESSABLE_ENTITY,
169  detail=msg,
170  )
171 
172  # Get latest submission date
173  valid_datetimes = [
174  form["lastSubmission"]
175  for form in odk_forms
176  if form["lastSubmission"] is not None
177  ]
178 
179  review_states = [
180  form["reviewStates"]
181  for form in odk_forms
182  if form["reviewStates"] is not None
183  ]
184 
185  last_submission = (
186  max(
187  valid_datetimes,
188  key=lambda x: datetime.strptime(x, "%Y-%m-%dT%H:%M:%S.%fZ"),
189  )
190  if valid_datetimes
191  else None
192  )
193  if not last_submission:
194  msg = f"Could not identify last submission for project ({project_id})"
195  log.warning(msg)
196  return HTTPException(
197  status_code=HTTPStatus.UNPROCESSABLE_ENTITY,
198  detail=msg,
199  )
200 
201  # Check if the file already exists in s3
202  s3_project_path = f"/{project.organisation_id}/{project_id}"
203  metadata_s3_path = f"/{s3_project_path}/submissions.meta.json"
204  try:
205  # Get the last submission date and review state from the metadata
206  data = get_obj_from_bucket(settings.S3_BUCKET_NAME, metadata_s3_path)
207  submission_metadata = json.loads(data.getvalue())
208 
209  zip_file_last_submission = submission_metadata["last_submission"]
210  zip_file_review_states = (
211  submission_metadata["review_states"]
212  if "review_states" in submission_metadata
213  else None
214  )
215 
216  # converting list into hash str to compare them easily
217  odk_review_state_hash = hashlib.sha256(
218  json.dumps(review_states).encode()
219  ).hexdigest()
220  s3_review_state_hash = hashlib.sha256(
221  json.dumps(zip_file_review_states).encode()
222  ).hexdigest()
223 
224  if (
225  last_submission <= zip_file_last_submission
226  and odk_review_state_hash == s3_review_state_hash

228  # Update background task status to COMPLETED

238  # Zip file is outdated, regenerate

240  "last_submission": last_submission,
241  "review_states": review_states,

267  add_obj_to_bucket(
268  settings.S3_BUCKET_NAME,
269  metadata_obj,
270  metadata_s3_path,
271  )

273  # Update background task status to COMPLETED
274  update_bg_task_sync = async_to_sync(

279  return True

329 async def get_submission_points(db: Session, project_id: int, task_id: Optional[int]):
330  """Get submission points for a project."""
331  project_info = await project_crud.get_project_by_id(db, project_id)
332  if not project_info:
333  raise HTTPException(status_code=404, detail="Project not found")
334 
335  odk_id = project_info.odkid
336  odk_credentials = await project_deps.get_odk_credentials(db, project_id)
337  xform = get_odk_form(odk_credentials)
338  db_xform = await project_deps.get_project_xform(db, project_id)
339 
340  response_file = xform.getSubmissionMedia(odk_id, db_xform.odk_form_id)
341  response_file_bytes = response_file.content
342 
343  try:
344  with zipfile.ZipFile(io.BytesIO(response_file_bytes), "r") as zip_ref:
345  csv_filenames = [f for f in zip_ref.namelist() if f.endswith(".csv")]
346  if not csv_filenames:
347  print("No CSV files found in the zip archive.")
348  return None
349 
350  csv_filename = csv_filenames[0]
351  with zip_ref.open(csv_filename) as csv_file:
352  csv_reader = csv.DictReader(io.TextIOWrapper(csv_file))
353  geometry = []
354 
355  for row in csv_reader:
356  if not task_id or int(row["all-task_id"]) == task_id:
357  latitude = row.get("warmup-Latitude")
358  longitude = row.get("warmup-Longitude")
359  if latitude and longitude:
360  point = (latitude, longitude)
361  geometry.append(
362  {
363  "type": "Feature",
364  "geometry": {"type": "Point", "coordinates": point},
365  }
366  )
367 
368  return geometry
369 
370  except zipfile.BadZipFile:
371  print("The file is not a valid zip file.")
372  return None

390 async def get_submissions_by_date(
391  db: Session, project_id: int, days: int, planned_task: Optional[int] = None
392 ):
393  """Get submissions by date.
394 
395  Fetches the submissions for a given project within a specified number of days.
396 
397  Args:
398  db (Session): The database session.
399  project_id (int): The ID of the project.
400  days (int): The number of days to consider for fetching submissions.
401  planned_task (int): Associated task id.
402 
403  Returns:
404  dict: A dictionary containing the submission counts for each date.
405 
406  Examples:
407  # Fetch submissions for project with ID 1 within the last 7 days
408  submissions = await get_submissions_by_date(db, 1, 7)
409  """
410  data = await get_submission_by_project(project_id, {}, db)
411 
412  end_dates = [
413  datetime.fromisoformat(entry["end"].split("+")[0])
414  for entry in data["value"]
415  if entry.get("end")
416  ]
417 
418  dates = [
419  date.strftime("%m/%d")
420  for date in end_dates
421  if datetime.now() - date <= timedelta(days=days)
422  ]
423 
424  submission_counts = Counter(sorted(dates))
425 
426  response = [
427  {"date": key, "count": value} for key, value in submission_counts.items()
428  ]
429  if planned_task:
430  count_dict = {}
431  cummulative_count = 0
432  for date, count in submission_counts.items():
433  cummulative_count += count
434  count_dict[date] = cummulative_count
435  response = [
436  {"date": key, "count": count_dict[key], "planned": planned_task}
437  for key, value in submission_counts.items()
438  ]
439 
440  return response
